-- 1. Selectează numele și adresa clienților împreună cu totalul comenzilor lor, dar doar pentru clienții din segmentul "AUTOMOBILE" care au comenzi peste 1000
-- 1. Select complex fara grupare

db.customer.aggregate([
    {
        $lookup: {
            from: "lineitem_orders",
            localField: "c_custkey",
            foreignField: "o_custkey",
            as: "orders"
        }
    },
    {
        $unwind: "$orders"
    },
    {
        $match: {
            "orders.o_totalprice": { $gt: 1000 },
            "c_mktsegment": "AUTOMOBILE"
        }
    },
    {
        $project: {
            _id: 0,
            c_name: 1,
            c_address: 1,
            "orders.o_totalprice": 1
        }
    }
]).pretty();

-- 2. Selectează numele și adresa furnizorilor care au un sold mai mare decât media tuturor furnizorilor și care aparțin unei națiuni al cărei nume începe cu "A".
-- 2. Select cu predicat dublu AND si subquery

var avgAcctBal = db.supplier.aggregate([
    {
        $group: {
            _id: null,
            avg_acctbal: { $avg: "$s_acctbal" }
        }
    }
]).toArray()[0].avg_acctbal;

db.supplier.aggregate([
    {
        $lookup: {
            from: "nation_region",
            localField: "s_nationkey",
            foreignField: "nation.n_nationkey",
            as: "nation_data"
        }
    },
    {
        $unwind: "$nation_data"
    },
    {
        $match: {
            "s_acctbal": { $gt: avgAcctBal }, 
            "nation_data.nation.n_name": { $regex: /^A/ }
        }
    },
    {
        $project: {
            _id: 0,
            s_name: 1,
            s_address: 1
        }
    }
]).pretty();

-- 14. Calculează prețul final ((l_extendedprice * (1 - l_discount) * (1 + l_tax))) pentru liniile de comandă între 1994 și 1995.

db.lineitem.aggregate([
    {
        $match: {
            l_shipdate: { $gte: "1994-01-01", $lte: "1995-12-31" } 
        }
    },
    {
        $project: {
            _id: 0,
            l_orderkey: 1,
            final_price: {
                $multiply: [
                    "$l_extendedprice",
                    { $subtract: [1, "$l_discount"] }, 
                    { $add: [1, "$l_tax"] }          
                ]
            }
        }
    }
]).pretty();

-- 18. Creează o coloană nouă care etichetează clienții ca 'VIP' dacă au soldul mai mare decât media tuturor clienților, altfel 'Regular'.

db.customer.aggregate([
    {
        $facet: {
            avg_acctbal_calc: [
                {
                    $group: {
                        _id: null,
                        avg_acctbal: { $avg: "$c_acctbal" } 
                    }
                }
            ],
            customers_data: [
                {
                    $project: {
                        _id: 0,
                        c_name: 1,
                        c_acctbal: 1
                    }
                }
            ]
        }
    },
    {
        $unwind: "$avg_acctbal_calc"
    },
    {
        $set: {
            avg_acctbal: "$avg_acctbal_calc.avg_acctbal"
        }
    },
    {
        $project: {
            avg_acctbal_calc: 0 
        }
    },
    {
        $unwind: "$customers_data"
    },
    {
        $addFields: {
            "customers_data.customer_status": {
                $cond: {
                    if: { $gt: ["$customers_data.c_acctbal", "$avg_acctbal"] },
                    then: "VIP",
                    else: "Regular"
                }
            }
        }
    },
    {
        $replaceRoot: { newRoot: "$customers_data" }
    }
]).pretty();

-- 198. Clienți care au făcut cea mai mare creștere a valorii comenzilor într-un an

db.customer.aggregate([
  // $lookup pentru a face JOIN între customer și orders
  {
    $lookup: {
      from: "lineitem_orders", // Corectăm pentru numele colecției corespunzătoare
      localField: "c_custkey",
      foreignField: "o_custkey",
      as: "orders"
    }
  },

  // $unwind pentru a desface array-ul orders
  { $unwind: "$orders" },

  // Adăugăm câmpul 'year' și 'total_spent' pentru fiecare comandă
  {
    $addFields: {
      year: { $year: { $dateFromString: { dateString: "$orders.o_orderdate" } } }, // Extragem anul
      total_spent: "$orders.o_totalprice"
    }
  },

  // Grupăm pe c_custkey și year pentru a calcula suma totală cheltuită pe an
  {
    $group: {
      _id: { c_name: "$c_name", c_custkey: "$c_custkey", year: "$year" },
      total_spent: { $sum: "$total_spent" }
    }
  },

  // Sortăm pentru a ajuta la calculul creșterii (creșterea va depinde de ordine)
  { $sort: { "_id.c_custkey": 1, "_id.year": 1 } },

  // Grupăm din nou per client pentru a adăuga un array cu anii și totalurile
  {
    $group: {
      _id: { c_name: "$_id.c_name", c_custkey: "$_id.c_custkey" },
      years: {
        $push: { year: "$_id.year", total_spent: "$total_spent" }
      }
    }
  },

  // Simulăm LAG și calculăm creșterea
  {
    $addFields: {
      years_with_growth: {
        $map: {
          input: "$years",
          as: "current_year",
          in: {
            $let: {
              vars: {
                prev_year: {
                  $cond: {
                    if: { $eq: [{ $size: "$years" }, 1] },  // Dacă există doar un an, prev_year va fi 0
                    then: 0,
                    else: {
                      $arrayElemAt: [
                        "$years",
                        { $subtract: [
                            { $indexOfArray: ["$years.year", "$$current_year.year"] },
                            1
                          ] 
                        }
                      ]
                    }
                  }
                }
              },
              in: {
                year: "$$current_year.year",
                total_spent: "$$current_year.total_spent",
                prev_year_spent: { $ifNull: ["$$prev_year.total_spent", 0] },
                growth: {
                  $subtract: [
                    "$$current_year.total_spent",
                    { $ifNull: ["$$prev_year.total_spent", 0] }
                  ]
                }
              }
            }
          }
        }
      }
    }
  },

  // Dezvăluim datele calculând creșterea pentru fiecare an al clientului
  { $unwind: "$years_with_growth" },

  // Proiectăm doar câmpurile necesare și eliminăm _id
  {
    $project: {
      c_name: "$_id.c_name",
      year: "$years_with_growth.year",
      total_spent: "$years_with_growth.total_spent",
      prev_year_spent: "$years_with_growth.prev_year_spent",
      growth: "$years_with_growth.growth",
      _id: 0  // Eliminăm câmpul _id
    }
  },

  // Sortăm după creșterea în ordine descrescătoare
  { $sort: { growth: -1 } }
])
